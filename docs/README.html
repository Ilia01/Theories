    You can convert these markdown files to HTML for better readability in a browser.

    Option 1: Using markdown-it (Node.js)

    # Install markdown-it
npm install -g markdown-it

# Convert single file
markdown-it 01-git-version-control.md > output.html

# Convert all files
for file in *.md; do
  markdown-it "$file" > "${file%.md}.html"
done

    Option 2: Using marked (Node.js)

    # Install marked
npm install -g marked

# Convert single file
marked -i 01-git-version-control.md -o output.html

# Convert all files
for file in *.md; do
  marked -i "$file" -o "${file%.md}.html"
done

    Option 3: Using Pandoc

    # Install pandoc (https://pandoc.org/installing.html)

# Convert with styling
pandoc 01-git-version-control.md -o output.html --standalone --css=style.css

# Convert all files
for file in *.md; do
  pandoc "$file" -o "${file%.md}.html" --standalone
done

    Option 4: Using Python (markdown package)

    import markdown
import os

def convert_md_to_html(md_file):
    with open(md_file, 'r', encoding='utf-8') as f:
        text = f.read()

    html = markdown.markdown(text, extensions=['tables', 'fenced_code'])

    output_file = md_file.replace('.md', '.html')
    with open(output_file, 'w', encoding='utf-8') as f:
        f.write(f"""
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>{os.path.basename(md_file)}</title>
    <style>
        body {{
            font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Helvetica, Arial, sans-serif;
            line-height: 1.6;
            max-width: 900px;
            margin: 0 auto;
            padding: 20px;
            background: #f5f5f5;
        }}
        code {{
            background: #f4f4f4;
            padding: 2px 6px;
            border-radius: 3px;
            font-family: 'Courier New', Courier, monospace;
        }}
        pre {{
            background: #2d2d2d;
            color: #f8f8f2;
            padding: 15px;
            border-radius: 5px;
            overflow-x: auto;
        }}
        pre code {{
            background: none;
            color: inherit;
        }}
        table {{
            border-collapse: collapse;
            width: 100%;
            margin: 20px 0;
        }}
        table, th, td {{
            border: 1px solid #ddd;
        }}
        th, td {{
            padding: 12px;
            text-align: left;
        }}
        th {{
            background-color: #4CAF50;
            color: white;
        }}
        tr:nth-child(even) {{
            background-color: #f2f2f2;
        }}
        h1, h2, h3 {{
            color: #333;
        }}
        h1 {{
            border-bottom: 2px solid #4CAF50;
            padding-bottom: 10px;
        }}
        h2 {{
            margin-top: 30px;
            border-bottom: 1px solid #ddd;
            padding-bottom: 5px;
        }}
        blockquote {{
            border-left: 4px solid #4CAF50;
            padding-left: 20px;
            color: #666;
            font-style: italic;
        }}
        a {{
            color: #4CAF50;
            text-decoration: none;
        }}
        a:hover {{
            text-decoration: underline;
        }}
    </style>
</head>
<body>
{html}
</body>
</html>
        """)

# Convert all markdown files
for file in os.listdir('.'):
    if file.endswith('.md') and file != 'README.md':
        convert_md_to_html(file)
        print(f"Converted {file}")
